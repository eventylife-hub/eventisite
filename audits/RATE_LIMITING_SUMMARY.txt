================================================================================
                              USAGE EXAMPLES
================================================================================

APPLYING THE DECORATOR:

import { RateLimit, RateLimitProfile } from '@/common/decorators/rate-limit.decorator';

@Controller('auth')
export class AuthController {
  @RateLimit(RateLimitProfile.AUTH)
  @Post('login')
  async login(@Body() dto: LoginDto) {
    // Limited to 5 requests per 60 seconds
  }
}

@Controller('checkout')
export class CheckoutController {
  @RateLimit(RateLimitProfile.PAYMENT)
  @Post(':id/payment')
  async createPaymentSession(@Param('id') groupId: string) {
    // Limited to 10 requests per 60 seconds
  }
}

@Controller('payments')
export class PaymentsController {
  @RateLimit(RateLimitProfile.WEBHOOK)
  @Post('webhook')
  async handleWebhook(@RawBody() body: Buffer) {
    // Limited to 200 requests per 60 seconds
  }
}
